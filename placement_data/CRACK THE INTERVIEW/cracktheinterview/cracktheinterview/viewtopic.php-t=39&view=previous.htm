#include <stdio.h>
#include "malloc.h"

#define TRUE 1
#define FALSE 0
#define BOOL unsigned char

typedef struct _NUMNODE
{
int Num ;
struct _NUMNODE *pNext ;
}NUMNODE ;

NUMNODE *pNumList = NULL ;
NUMNODE *pNumSortList = NULL ;

BOOL InsertNumIntoList ( int Num )
{
NUMNODE *pTempNode = NULL ;

pTempNode = (NUMNODE*) malloc ( sizeof(NUMNODE) );
if ( !pTempNode )
return(FALSE);

pTempNode->Num = Num ;
pTempNode->pNext = pNumList ;

// Store items in reverse list, doesnt matter we are going to sort later
pNumList = pTempNode ;

return(TRUE);
}

// Bubble sort. We can use other methods also. In this method, i am doing Node data exchange.
BOOL SortLinkedList ( )
{
NUMNODE *pTmpList = pNumList ;
NUMNODE *pTmpList1 = NULL ;
int TmpNum = 0 ;

while ( pTmpList && pTmpList->pNext != NULL )
{
pTmpList1 = pTmpList->pNext ;
while ( pTmpList1 != NULL )
{
if ( pTmpList->Num > pTmpList1->Num )
{
TmpNum = pTmpList1->Num ;
pTmpList1->Num = pTmpList->Num ;
pTmpList->Num = TmpNum ;
}

pTmpList1 = pTmpList1->pNext ;
}
pTmpList = pTmpList->pNext ;
}

return(TRUE);
}

NUMNODE *GetMostOccurance ( int *pOccurCount, NUMNODE **pPrevNode )
{
NUMNODE *pTmpList = NULL ;
NUMNODE *pTmpList1 = NULL ;
NUMNODE *pPrevNode1 = NULL ;
NUMNODE *pMaxOccNodeStart = NULL ;
int CurNum = 0 ;
int Count = 0 ;
int PrevCount = 0 ;
*pOccurCount = 0 ;

pTmpList = pNumList ;

while ( pTmpList != NULL )
{
pTmpList1 = pTmpList ;
Count = 0 ;
CurNum = pTmpList->Num ;

// Identify which number has most occurance in list
while ( pTmpList1 != NULL )
{
if ( pTmpList1->Num == CurNum )
Count ++ ;
else
break ;
pTmpList1 = pTmpList1->pNext ;
}

if ( PrevCount < Count )
{
PrevCount = Count ;
*pPrevNode = pPrevNode1 ;
pMaxOccNodeStart = pTmpList ;
}

pPrevNode1 = pTmpList ;
pTmpList = pTmpList1 ;
}

*pOccurCount = PrevCount ;

return(pMaxOccNodeStart);
}

BOOL SortBasedOnMostOccur ( )
{
int MostOccurCount = 0 ;
NUMNODE *pTmpList = NULL ;
NUMNODE *pPrevNode = NULL ;

while ( TRUE )
{
pTmpList = GetMostOccurance ( &MostOccurCount, &pPrevNode );
if ( pTmpList == NULL )
break ;

NUMNODE *pTmpNumSortList = NULL ;

// First time
if ( pNumSortList == NULL )
{
pNumSortList = pTmpList ;
pTmpNumSortList = pNumSortList ;
}
else
{
pTmpNumSortList = pNumSortList ;

while ( pTmpNumSortList->pNext != NULL )
pTmpNumSortList = pTmpNumSortList->pNext ;

pTmpNumSortList->pNext = pTmpList ;

pTmpNumSortList = pTmpNumSortList->pNext ;
}

for ( int Index = 0 ; Index < MostOccurCount - 1 ; Index ++ )
{
pTmpList = pTmpList->pNext ;

pTmpNumSortList = pTmpNumSortList->pNext ;
}

// We need to update root node itself
if ( pPrevNode == NULL )
pNumList = pTmpList->pNext ;
else
{
int Num ;
Num = pPrevNode->Num ;
while ( pPrevNode->pNext != NULL )
{
if ( pPrevNode->pNext->Num != Num )
break;

pPrevNode = pPrevNode->pNext ;
}
pPrevNode->pNext = pTmpList->pNext ;
}

pTmpNumSortList->pNext = NULL ;

}

return(TRUE);
}

BOOL DeleteLinkedList ( )
{
NUMNODE *pTmpNode = NULL ;

while ( pNumList != NULL )
{
